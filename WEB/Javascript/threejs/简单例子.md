# 简单例子

```vue
<template>
  <div id="container"></div>
</template>

<script>
import * as Three from 'three'

export default {
  name: 'HelloWorld',
  data() {
    return {
      camera: null,
      scene: null,
      renderer: null,
      mesh: null
    }
  },
  methods: {
    // 初始化
    init () {
      let container = document.getElementById('container');

      // 声明相机,这里使用的是透视相机 (视野角度,长宽比,近截面,远截面)
      this.camera = new Three.PerspectiveCamera(70, container.clientWidth/container.clientHeight, 0.01, 10);
      this.camera.position.z = 1;
      this.scene = new Three.Scene();


      this.light = new Three.AmbientLight( 0x404040 ); // soft white light

      // 声明一个立方体,BoxGeometry（立方体）
      let geometry = new Three.BoxGeometry(0.2, 0.2, 0.4);
      // 材质
      let material = new Three.MeshNormalMaterial();

      //  网格,包含一个几何体以及作用在此几何体上的材质，我们可以直接将网格对象放入到我们的场景中，并让它在场景中自由移动。
      this.mesh = new Three.Mesh(geometry, material);

      this.scene.add(this.mesh, this.light);

      this.renderer = new Three.WebGLRenderer(
        {
          // 抗锯齿打开
          antialias: true
        }
      );
      // 设置一个渲染器的尺寸
      this.renderer.setSize(container.clientWidth, container.clientHeight);
      // 我们将renderer（渲染器）的dom元素（renderer.domElement）添加到我们的HTML文档中
      container.appendChild(this.renderer.domElement);

      this.mesh.rotation.x += 1.01;
      this.mesh.rotation.y += 1.02;
      this.renderer.render(this.scene, this.camera);
    },

    // 动画
    animate () {
      requestAnimationFrame(this.animate);
      this.mesh.rotation.x += 0.002;
      this.mesh.rotation.y += 0.004;
      this.renderer.render(this.scene, this.camera);
    }
  },
  mounted() {
    this.init();
    // this.animate();
  }
}
</script>

<style scoped>
#container{
  width: 100%;
  height: 100%;
}
</style>

```
