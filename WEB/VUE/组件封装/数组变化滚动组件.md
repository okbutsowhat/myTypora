# 数字滚动组件

```js
<template>
  <div class="digitalScroll">
    <ul
      v-for="(item, index) in splitNumber"
      :style="`transform: translateY(-${item * boxHeight}px);`"
    >
      <li>0</li>
      <li
        v-for="value in 9">
        {{value}}
      </li>
    </ul>
  </div>
</template>

<script>
export default {
  name: "digitalScroll",
  props: {
    num: {
      type: Number,
      default: 123456
    },
    boxHeight: {
      type: Number,
      default: 72
    }
  },
  computed: {
    splitNumber() {
      return [...this.num + ''].map(n => +n)
    }
  },
  mounted() {
    console.log(this.splitNumber);
  }
}
</script>

<style scoped>
.digitalScroll{
  display: flex;
  justify-content: flex-end;
  align-items: center;
}
.digitalScroll ul{
  transition: all 0.5s ease-out;
}
</style>

```
